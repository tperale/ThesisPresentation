\begin{frame}{My Approach}

  \begin{itemize}
    \item IPv6 enabled
    \item Power efficient
    \item Reliable
  \end{itemize}
  \item Standardized protocol
  \begin{itemize}
    \item Routing Protocol for Low-Power and Lossy Networks (RPL)
    \item Time Slotted Channel Hopping (TSCH)
  \end{itemize}
